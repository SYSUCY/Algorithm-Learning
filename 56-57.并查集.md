# 并查集

## 使用场景

并查集的使用是如下的场景
1. 一开始每个元素都拥有自己的集合，在自己的集合里只有这个元素自己
2. find(i)：查找i所在集合的代表元素，代表元素来代表i所在的集合
3. bool isSameSet(a, b)：判断a和b在不在一个集合里
4. void union(a, b)：a所在集合所有元素 与 b所在集合所有元素 合并成一个集合
5. 各种操作单次调用的均摊时间复杂度O(1)

## 并查集的优化
1. 扁平化（**一定要做**）
2. 小挂大（可以不做）

## 并查集的小扩展
可以定制信息：并查集目前有多少个集合，以及给每个集合打上标签信息

## 模板题
### 扁平化 + 小挂大
测试链接：https://www.nowcoder.com/practice/e7ed657974934a30b2010046536a5372
```c++
#include<cstdio>
const int MAXN = 1000005;
int father[MAXN];
int size[MAXN];
int stack[MAXN];
int stackSize;
int N;

//初始化并查集
void build() {
    for (int i = 1; i <= N; i++) {
        father[i] = i;
        size[i] = 1;
    }
}

//找到i元素所在集合的代表元素
int find(int i) {
    stackSize = 0;
    while (i != father[i]) {
        stack[stackSize++] = i; //在跳转到父节点之前就要收集了
        i = father[i];
    }
    while (stackSize > 0)
        father[stack[--stackSize]] = i; //沿途节点的父亲全部变成集合的代表元素
    return i;
}

bool isSameSet(int a, int b) {
    return find(a) == find(b);
}

void unionTwoDifferentSet(int a, int b) {
    int fa = find(a);
    int fb = find(b);
    //不在同一个集合才合并
    if (fa != fb) {
        if (size[fa] > size[fb]) {
            size[fa] += size[fb];
            father[fb] = fa;
        } else {
            size[fb] += size[fa];
            father[fa] = fb;
        }
    }
}

int main() {
    int M, opt, x, y;
    scanf("%d%d", &N, &M);
    build();
    while (M--) {
        scanf("%d%d%d", &opt, &x, &y);
        switch (opt) {
            case 1:
                printf("%s\n", isSameSet(x, y) ? "Yes" : "No");
                break;
            case 2:
                unionTwoDifferentSet(x, y);
                break;
        }
    }

    return 0;
}
```

### 省略小挂大 + 用递归实现扁平化（更常用）
测试链接：https://www.luogu.com.cn/problem/P3367
```c++
#include<cstdio>
const int MAXN = 10005;
int father[MAXN];
int N;

void build() {
	for (int i = 1; i <= N; i++) 
		father[i] = i;
}

int find(int i) {
	if (i != father[i])	//找到祖先
		father[i] = find(father[i]);	//扁平化
	return father[i];
}

bool isSameSet(int x, int y) {
	return find(x) == find(y);
}

void unionTwoDifferentSet(int x, int y) {
	//两种情况
	//情况一：x,y是同一个集合，代表元素自己指向自己，没变化
	//情况二：x,y不是同一个集合，x所在集合代表元素指向y所在集合代表元素
	father[find(x)] = find(y);
}

int main() {
	int M;
	scanf("%d%d", &N, &M);
	build();
	int opt, x, y;
	while (M--) {
		scanf("%d%d%d", &opt, &x, &y);
		switch (opt) {
			case 1:
				unionTwoDifferentSet(x, y);
				break;
			case 2:
				printf("%c\n", isSameSet(x, y) ? 'Y' : 'N');
				break;
		}
	}

	return 0;
}
```

### 题目1：情侣牵手
测试链接：https://leetcode.cn/problems/couples-holding-hands/
```c++
//考虑集合中混合的情侣对数
//一对：交换0次座位
//两对：交换1次座位
//三对：交换2次座位
//...
//m对：交换m-1次座位
//假设总共有k个混合集合，mi为第i个混合集合中情侣对数
//交换次数 =(m1 - 1) + (m2 - 1) + ... (mk - 1) 
//        = (m1 + m2 + ... + mk) - k
//        = n - k 
class Solution {
private:
    static const int MAXN = 30; //最大情侣对数
    static int father[MAXN];
    int sets;
    void build(int n){
        sets = n;   //混合集合个数，开始设置为总共的情侣对数
        for(int i = 0; i < n; i++)
            father[i] = i;
    }
    int find(int i){
        if(i != father[i])
            father[i] = find(father[i]);
        return father[i];
    }
    void unionTwoDifferentSet(int x, int y){
        int fx = find(x);
        int fy = find(y);
        if(fx != fy){
            father[fx] = fy;
            sets--; //两个混合集合合并了，混合集合总数减一
        }
    }
public: 
    int minSwapsCouples(vector<int>& row) {
        int n = row.size(); //总共的人数 
        build(n / 2);
        //一对一对的遍历，如果不是情侣（在集合中的代表元素不一样），就合并到一个混合集合中
        for(int i = 0; i < n; i += 2)
            unionTwoDifferentSet(row[i] / 2, row[i + 1] / 2);
        return n / 2 - sets;
    }
};
int Solution::father[MAXN];
```

### 题目2：相似字符串组
测试链接：https://leetcode.cn/problems/similar-string-groups/
```c++
class Solution {
private:
    static const int MAXN = 305;
    static int father[MAXN];
    int sets;
    //初始假设所有的字符串相互之间都不相似
    void build(int n){
        sets = n;
        for(int i = 0; i < n; i++)
            father[i] = i;
    }
    int find(int i){
        if(i != father[i])
            father[i] = find(father[i]);
        return father[i];
    }
    void unionTwoDifferentSet(int x, int y){
        int fx = find(x);
        int fy = find(y);
        if(fx != fy){
            father[fx] = fy;
            sets--;
        }
    }
public:
    int numSimilarGroups(vector<string>& strs) {
        int n = strs.size();
        build(n);
        //看字符串两两之间是否相似，相似就合并为一组
        int len = strs[0].size();   //字符串的长度
        for(int i = 0; i < n - 1; i++){
            for(int j = i + 1; j < n; j++){
                int diff = 0; //不同的字符
                for(int k = 0; k < len && diff < 3; k++)
                    diff = strs[i][k] != strs[j][k] ? diff + 1 : diff;
                if(diff == 0 || diff == 2)  //两个字符串相似，合并为一个关联组
                    unionTwoDifferentSet(i, j);
            }
        }
        return sets;
    }
};
int Solution::father[MAXN];
```

### 题目3：岛屿数量
测试链接：https://leetcode.cn/problems/number-of-islands/
```c++
class Solution {
private:
    static const int MAXN = 90005;
    static int father[MAXN];
    int sets;
    int rows, cols;

    int index(int row, int col) { return row * cols + col; }
    void build(vector<vector<char>>& grid) {
        sets = 0;
        for (int row = 0, i; row < rows; row++) {
            for (int col = 0; col < cols; col++) {
                if (grid[row][col] == '1') {
                    i = index(row, col);
                    father[i] = i;
                    sets++;
                }
            }
        }
    }
    int find(int i) {
        if (i != father[i])
            father[i] = find(father[i]);
        return father[i];
    }
    void unionTwoDifferentSet(int x, int y) {
        int fx = find(x);
        int fy = find(y);
        if (fx != fy) {
            father[fx] = fy;
            sets--;
        }
    }

public:
    int numIslands(vector<vector<char>>& grid) {
        rows = grid.size();
        cols = grid[0].size();
        build(grid); // 初始化并查集，开始假设每个岛屿互补相连
        for (int row = 0; row < rows; row++) {
            for (int col = 0; col < cols; col++) {
                if (grid[row][col] == '1') {
                    if (row > 0 && grid[row - 1][col] == '1')
                        unionTwoDifferentSet(index(row - 1, col),
                                             index(row, col));
                    if (col > 0 && grid[row][col - 1] == '1')
                        unionTwoDifferentSet(index(row, col - 1),
                                             index(row, col));
                }
            }
        }
        return sets;
    }
};
int Solution::father[MAXN];
```

### 题目4：移除最多的同行或同列的石头
测试链接：https://leetcode.cn/problems/most-stones-removed-with-same-row-or-column/
```c++
#include<unordered_map>
class Solution {
private:
    static const int MAXN = 1005;
    static int father[MAXN];
    int sets;
    static unordered_map<int, int> rowIndexMap;
    static unordered_map<int, int> colIndexMap;

    void build(int n){
        rowIndexMap.clear();
        colIndexMap.clear();
        sets = n;
        for(int i = 0; i < n; i++)
            father[i] = i;
    }

    int find(int i){
        if(i != father[i])
            father[i] = find(father[i]);
        return father[i];
    }

    void unionTwoDifferentSet(int x, int y){
        int fx = find(x);
        int fy = find(y);
        if(fx != fy){
            father[fx] = fy;
            sets--;
        }
    }
public:
    int removeStones(vector<vector<int>>& stones) {
        int n = stones.size();
        build(n);
        int row, col;
        for(int i = 0; i < n; i++){
            row = stones[i][0];
            col = stones[i][1];
            if(rowIndexMap.count(row))  //之前这行有石头
                unionTwoDifferentSet(i, rowIndexMap[row]);
            else
                rowIndexMap[row] = i;
            if(colIndexMap.count(col))  //之前这列有石头
                unionTwoDifferentSet(i, colIndexMap[col]);
            else
                colIndexMap[col] = i;
        }
        return n - sets;    //可以移除的石头 = 所有石头 - 剩下的石头
    }
};

int Solution::father[MAXN];
unordered_map<int, int> Solution::rowIndexMap;
unordered_map<int, int> Solution::colIndexMap;
```

### 题目5：找出知晓秘密的所有专家
测试链接：https://leetcode.cn/problems/find-all-people-with-secret/
```c++
#include <algorithm>
class Solution {
private:
    static const int MAXN = 100005;
    static int father[MAXN];
    static bool knowSecret[MAXN];

    void build(int n, int firstPerson){
        for(int i = 0; i < n; i++){
            father[i] = i;
            knowSecret[i] = false;
        }
        //开始就有一个知道秘密的集合了
        father[firstPerson] = 0;
        knowSecret[0] = true;
    }

    int find(int i){
        if(i != father[i])
            father[i] = find(father[i]);
        return father[i];
    }

    void unionTwoDifferentSet(int x, int y){
        int fx = find(x);
        int fy = find(y);
        if(fx != fy){
            father[fx] = fy;
            //相当于knowSecret[fy] = knowSecret[fy] | knowSecret[fx]
            //一个人知道就都知道了
            knowSecret[fy] |= knowSecret[fx]; 
        }
    }
public:
    vector<int> findAllPeople(int n, vector<vector<int>>& meetings, int firstPerson) {
        sort(meetings.begin(), meetings.end(), [](vector<int>& m1, vector<int>& m2) {
                 return m1[2] < m2[2];
        }); // 会议按时间排序
        build(n, firstPerson);
        int m = meetings.size();    //会议数量
        //一次处理同时发生的会议
        for(int l = 0, r = 0; l < m; l = r + 1){
            r = l;
            //开会时间一样就后移
            while(r + 1 < m && meetings[r + 1][2] == meetings[r][2])
                r++;
            //同一时间开会的专家消息互通一下
            for(int i = l; i <= r; i++)
                unionTwoDifferentSet(meetings[i][0], meetings[i][1]);
            //开完会还不知道秘密的专家重新变成自己一个集合
            for(int i = l, cur; i <= r; i++){
                //代表专家都不知道就是真不知道了
                //每个会议涉及两个专家，都要看一下
                for(int j = 0; j <= 1; j++){
                    cur = meetings[i][j];
                    if(!knowSecret[find(cur)])
                    father[cur] = cur;
                }          
            }
        }
        vector<int> ans;
        for(int i = 0; i < n; i++){
            if(knowSecret[find(i)])
                ans.push_back(i);
        }
        return ans;
    }
};

int Solution::father[MAXN];
bool Solution::knowSecret[MAXN];
```

### 题目6：好路径的数目
测试链接：https://leetcode.cn/problems/number-of-good-paths/
```c++
#include<algorithm>
class Solution {
private:
    static const int MAXN = 30005;
    static int father[MAXN]; //代表节点维护代表集合成最大值
    static int maxSize[MAXN];
    int goodPaths;
    
    void build(int n){
        goodPaths = n;
        for(int i = 0; i < n; i++){
            father[i] = i;
            maxSize[i] = 1;
        }
    }

    int find(int i){
        if(i != father[i])
            father[i] = find(father[i]);
        return father[i];
    }

    void unionTwoDifferentSet(vector<int>& vals, int x, int y){
        int fx = find(x);
        int fy = find(y);
        if(fx != fy){
            if(vals[fx] > vals[fy])
                father[fy] = fx;
            else if(vals[fx] < vals[fy])
                father[fx] = fy;
            else{   //两个节点集合的最大值一样
                goodPaths += maxSize[fx] * maxSize[fy];
                father[fx] = fy;
                maxSize[fy] += maxSize[fx]; 
            }
        }
    }
public:
    int numberOfGoodPaths(vector<int>& vals, vector<vector<int>>& edges) {
        //路径按节点最大值排序
        sort(edges.begin(), edges.end(), [&vals](vector<int> &edge1, vector<int> &edge2){
            return max(vals[edge1[0]], vals[edge1[1]]) < max(vals[edge2[0]], vals[edge2[1]]);
        });
        int n = vals.size();
        build(n);
        for(vector<int> &edge : edges)
            unionTwoDifferentSet(vals, edge[0], edge[1]);
        return goodPaths;
    }
};
int Solution::father[MAXN];
int Solution::maxSize[MAXN];
```

### 题目7：尽量减少恶意软件的传播
测试链接：https://leetcode.cn/problems/minimize-malware-spread-ii/
```c++
#include<algorithm>
class Solution {
private:
    static const int MAXN = 305;
    static bool virus[MAXN];
    static int cnts[MAXN];
    static int infect[MAXN];    //-1代表没被感染，-2代表被超过2个恶意软件感染（没救了）
    static int father[MAXN];
    static int size[MAXN];

    void build(int n, vector<int>& initial){
        for(int i = 0; i < n; i++){
            virus[i] = false;
            cnts[i] = 0;
            infect[i] = -1;
            father[i] = i;
            size[i] = 1;
        }
        //初始化恶意软件位置，便于后续处理
        for(int bad : initial)
            virus[bad] = true;
    }

    int find(int i){
        if(i != father[i])
            father[i] = find(father[i]);
        return father[i];
    }

    void unionTwoDifferentSet(int x, int y){
        int fx = find(x);
        int fy = find(y);
        if(fx != fy){
            father[fx] = fy;
            size[fy] += size[fx];
        }
    }
public:
    int minMalwareSpread(vector<vector<int>>& graph, vector<int>& initial) {
        int n = graph.size();
        build(n, initial);
        //将非感染节点能合并的合并成子网络
        for(int i = 0; i < n; i++){
            if(virus[i])
                continue;
            for(int j = 0; j < n; j++){
                if(virus[j])
                    continue;
                if(graph[i][j] == 1)
                    unionTwoDifferentSet(i, j);
            }
        }
        //给每个集合设置病毒源（唯一才算，不唯一病毒源无法抢救）
        for(int bad : initial){
            for(int i = 0; i < n; i++){
                //感染节点不能是毒点
                if(virus[i])
                    continue;
                if(graph[bad][i] == 1){
                    //第一次被感染
                    int fi = find(i);
                    if(infect[fi] == -1)
                        infect[fi] = bad;
                    else if(infect[fi] != bad) //被其他初始毒点感染过了
                        infect[fi] = -2;
                }
            }
        }
        //统计数据，每个集合的唯一病毒源加上集合个数
        for(int i = 0; i < n; i++){
            //保证一个集合只取一次
            if(i == find(i) && infect[i] >= 0)
                cnts[infect[i]] += size[i];
        }
        sort(initial.begin(), initial.end());
        int ans = initial[0];
        for(int bad : initial){
            if(cnts[bad] > cnts[ans])
                ans = bad;
        }
        return ans;
    }
};

bool Solution::virus[MAXN];
int Solution::cnts[MAXN];
int Solution::infect[MAXN];    //-1代表没被感染，-2代表被超过2个恶意软件感染（没救了）
int Solution::father[MAXN];
int Solution::size[MAXN];
```
